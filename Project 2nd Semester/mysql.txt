CREATE DATABASE HOTEL;
USE HOTEL;

CREATE TABLE PI(
NAME varchar(32),
GENDER VARCHAR(10),
ADDRESS varchar(40),
AGE varchar(40),
CNIC VARCHAR(32),
PHONE VARCHAR(11),
ROOM_NO INT,
RESERVATION_DATE date,
HOW INT,
AMOUNT INT,
P_METHOD VARCHAR(30),
CARD_NO VARCHAR(30),
primary key(CNIC));
